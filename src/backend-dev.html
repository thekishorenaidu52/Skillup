1. Server-side Programming

Learn server-side programming concepts and implementation using popular technologies.
Node.js Fundamentals
// Basic Node.js Server const http = require('http'); const server = http.createServer((req, res) => { res.writeHead(200, {'Content-Type': 'text/plain'}); res.end('Hello World\n'); }); server.listen(3000, 'localhost', () => { console.log('Server running at http://localhost:3000/'); });
Express.js Framework
// Express.js REST API const express = require('express'); const app = express(); app.use(express.json()); app.get('/api/users', (req, res) => { res.json({ users: [] }); }); app.post('/api/users', (req, res) => { // Create new user res.status(201).json({ message: 'User created' }); }); app.listen(3000, () => { console.log('API Server running on port 3000'); });

2. Database Management

SQL Databases
-- SQL Database Operations CREATE TABLE users ( id INT PRIMARY KEY AUTO_INCREMENT, username VARCHAR(50) UNIQUE NOT NULL, email VARCHAR(100) UNIQUE NOT NULL, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); -- Query Examples SELECT * FROM users WHERE created_at >= DATE_SUB(NOW(), INTERVAL 7 DAY); INSERT INTO users (username, email) VALUES ('john_doe', 'john@example.com');
MongoDB NoSQL
// MongoDB Operations const mongoose = require('mongoose'); const userSchema = new mongoose.Schema({ username: { type: String, required: true, unique: true }, email: { type: String, required: true, unique: true }, createdAt: { type: Date, default: Date.now } }); const User = mongoose.model('User', userSchema); // Create user await User.create({ username: 'john_doe', email: 'john@example.com' });

Authentication & Authorization
// JWT Authentication const jwt = require('jsonwebtoken'); const authenticateToken = (req, res, next) => { const token = req.headers['authorization']?.split(' ')[1]; if (!token) return res.sendStatus(401); jwt.verify(token, process.env.JWT_SECRET, (err, user) => { if (err) return res.sendStatus(403); req.user = user; next(); }); };

4. Security Best Practices

Input Validation
// Input Validation Example const { body, validationResult } = require('express-validator'); app.post('/api/users', body('email').isEmail(), body('password').isLength({ min: 8 }), (req, res) => { const errors = validationResult(req); if (!errors.isEmpty()) { return res.status(400).json({ errors: errors.array() }); } // Process valid input });
Security Checklist:
  • Implement proper input validation
  • Use HTTPS for all communications
  • Hash passwords using bcrypt
  • Implement rate limiting
  • Use security headers
